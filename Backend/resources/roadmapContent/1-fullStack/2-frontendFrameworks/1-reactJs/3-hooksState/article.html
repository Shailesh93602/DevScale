React State & Core Hooks

Hooks are functions that let you "hook into" React state and lifecycle features from functional components. They were introduced in React 16.8 and are now the standard way to write React.

useState — Local Component State

import { useState } from "react";

function Counter() {
  const [count, setCount] = useState(0); // [state, setter]

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>+</button>
      <button onClick={() => setCount(count - 1)}>-</button>
      <button onClick={() => setCount(0)}>Reset</button>
    </div>
  );
}

// Functional updater pattern (safe for async updates)
setCount(prev => prev + 1);

State with Objects & Arrays

// Always create new references — never mutate state directly!
const [user, setUser] = useState({ name: "Alice", age: 30 });

// ❌ Wrong: mutates state directly
user.name = "Bob"; setUser(user);

// ✅ Correct: spread to create new object
setUser(prev => ({ ...prev, name: "Bob" }));

// Arrays
const [items, setItems] = useState(["Apple"]);
setItems(prev => [...prev, "Banana"]);      // Add
setItems(prev => prev.filter(i => i !== "Apple")); // Remove

useEffect — Side Effects

import { useEffect, useState } from "react";

function UserProfile({ userId }: { userId: number }) {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    setLoading(true);
    fetch(`/api/users/${userId}`)
      .then(r => r.json())
      .then(data => { setUser(data); setLoading(false); });

    // Cleanup function (runs on unmount or before next effect)
    return () => { /* cancel requests, clear timers */ };
  }, [userId]); // Dependency array — re-runs when userId changes

  if (loading) return <p>Loading...</p>;
  return <h2>{user?.name}</h2>;
}

useEffect Dependency Array

useEffect(() => { /* ... */ });          // Runs after EVERY render
useEffect(() => { /* ... */ }, []);      // Runs ONCE (on mount)
useEffect(() => { /* ... */ }, [id]);    // Runs when `id` changes

useRef — Persistent Values Without Re-render

import { useRef } from "react";

function Timer() {
  const timerRef = useRef<NodeJS.Timeout | null>(null);
  const inputRef = useRef<HTMLInputElement>(null);

  const start = () => {
    timerRef.current = setInterval(() => console.log("tick"), 1000);
  };
  const stop = () => clearInterval(timerRef.current!);

  // Accessing DOM elements directly
  const focusInput = () => inputRef.current?.focus();

  return <input ref={inputRef} />;
}

useContext — Avoid Prop Drilling

const ThemeContext = React.createContext("light");

function App() {
  return (
    <ThemeContext.Provider value="dark">
      <Toolbar />        {/* No need to pass theme down */}
    </ThemeContext.Provider>
  );
}

function ThemedButton() {
  const theme = useContext(ThemeContext); // "dark"
  return <button className={`btn-${theme}`}>Click</button>;
}

Custom Hooks

// Encapsulate reusable stateful logic
function useLocalStorage<T>(key: string, initial: T) {
  const [value, setValue] = useState<T>(() => {
    const stored = localStorage.getItem(key);
    return stored ? JSON.parse(stored) : initial;
  });

  const set = (newValue: T) => {
    setValue(newValue);
    localStorage.setItem(key, JSON.stringify(newValue));
  };

  return [value, set] as const;
}

// Usage
const [theme, setTheme] = useLocalStorage("theme", "light");
