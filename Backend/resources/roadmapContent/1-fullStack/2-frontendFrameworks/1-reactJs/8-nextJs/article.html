Next.js App Router & Server Components

Next.js App Router (introduced in v13) is a paradigm shift in React development. It brings React Server Components (RSC) to the mainstream, enabling components to fetch data and render on the server without shipping JavaScript to the client.

App Router File Conventions

app/
  ├── layout.tsx        # Root layout (wraps all pages)
  ├── page.tsx          # Home route: /
  ├── loading.tsx       # Automatic loading UI for suspense
  ├── error.tsx         # Error boundary for this route
  ├── not-found.tsx     # 404 handler
  ├── dashboard/
  │   ├── layout.tsx    # Dashboard nested layout
  │   ├── page.tsx      # /dashboard route
  │   └── settings/
  │       └── page.tsx  # /dashboard/settings route
  └── api/
      └── users/
          └── route.ts  # API route: GET/POST /api/users

Server Components vs Client Components

// Server Component (default) — runs on server, no JS sent to client
// Can: fetch data directly, access DB, read env vars
// Cannot: use useState, useEffect, event handlers, browser APIs

// app/products/page.tsx
async function ProductsPage() {
  // Direct DB/API call — no useEffect, no loading state
  const products = await fetch('https://api.example.com/products', {
    next: { revalidate: 3600 } // cache for 1 hour
  }).then(r => r.json());

  return (
    <main>
      {products.map(p => (
        <ProductCard key={p.id} product={p} />
      ))}
    </main>
  );
}

// Client Component — runs in browser, can use hooks
// 'use client' directive at the top of the file

'use client';

import { useState } from 'react';

function AddToCartButton({ productId }) {
  const [added, setAdded] = useState(false);

  return (
    <button onClick={() => setAdded(true)}>
      {added ? '✓ Added' : 'Add to Cart'}
    </button>
  );
}

Server Actions

Server Actions allow you to run server-side code directly from form submissions or event handlers — no API route needed:

'use server';

// actions/products.ts
export async function createProduct(formData: FormData) {
  const name = formData.get('name') as string;
  const price = Number(formData.get('price'));

  // Direct DB call — runs on server
  await db.product.create({ data: { name, price } });
  revalidatePath('/products'); // invalidate Next.js cache
}

// In a Server Component (no 'use client' needed!)
import { createProduct } from '@/actions/products';

function NewProductForm() {
  return (
    <form action={createProduct}>
      <input name="name" placeholder="Product name" />
      <input name="price" type="number" />
      <button type="submit">Create Product</button>
    </form>
  );
}

Data Fetching Strategies

// Static (build time) — fastest, no revalidation
const data = await fetch(url, { cache: 'force-cache' });

// Incremental Static Regeneration — revalidate every N seconds
const data = await fetch(url, { next: { revalidate: 60 } });

// Dynamic (every request) — always fresh
const data = await fetch(url, { cache: 'no-store' });

// Tag-based revalidation
const data = await fetch(url, { next: { tags: ['products'] } });
// Then revalidate programmatically:
revalidateTag('products');

Parallel Data Fetching with Suspense Streaming

// app/dashboard/page.tsx
import { Suspense } from 'react';

async function Dashboard() {
  return (
    <div>
      <Suspense fallback={<RevenueCardSkeleton />}>
        <RevenueCard />  {/* fetches its own data */}
      </Suspense>
      <Suspense fallback={<LatestInvoicesSkeleton />}>
        <LatestInvoices /> {/* fetches independently */}
      </Suspense>
    </div>
  );
}
// Result: both sections load in parallel and stream in as they resolve

Dynamic Routes & generateStaticParams

// app/blog/[slug]/page.tsx
export async function generateStaticParams() {
  const posts = await fetch('/api/posts').then(r => r.json());
  return posts.map(post => ({ slug: post.slug }));
}

async function BlogPost({ params }: { params: { slug: string } }) {
  const post = await fetch(`/api/posts/${params.slug}`).then(r => r.json());
  return <article><h1>{post.title}</h1>...</article>;
}

Best Practices

  • Default to Server Components — only add 'use client' when you need interactivity or browser APIs.
  • Push Client Components to the leaves of the component tree.
  • Use parallel routes with Suspense to avoid waterfall data fetching.
  • Use Server Actions for form mutations instead of separate API routes.
  • Use generateStaticParams to pre-render dynamic routes at build time.
