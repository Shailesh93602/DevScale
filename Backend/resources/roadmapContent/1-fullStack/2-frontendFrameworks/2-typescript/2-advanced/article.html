Advanced TypeScript

These advanced features are what make TypeScript truly powerful — generics, utility types, and conditional types let you write highly reusable, type-safe abstractions.

Generics

// Generic function — works with any type
function identity<T>(arg: T): T { return arg; }
identity<string>("hello"); // string
identity<number>(42);      // number

// Generic interface
interface ApiResponse<T> {
  data: T;
  status: number;
  message: string;
}

type UserResponse = ApiResponse<User>;
type PostsResponse = ApiResponse<Post[]>;

// Generic constraints
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}
getProperty({ name: "Alice", age: 30 }, "name"); // ✅ "Alice"
getProperty({ name: "Alice", age: 30 }, "email"); // ❌ compile error

Built-in Utility Types

interface User { id: number; name: string; email: string; role: string; }

Partial<User>        // All props optional { id?: number; name?: string; ... }
Required<User>       // All props required
Readonly<User>       // All props read-only
Pick<User, "name" | "email">  // Only selected props
Omit<User, "id">             // All except excluded props
Record<string, number>        // { [key: string]: number }

// Practical examples
type UpdateUserDto = Partial<Omit<User, "id">>;
type CreateUserDto = Omit<User, "id">;
type UserSummary = Pick<User, "id" | "name">;

Conditional Types

type IsString<T> = T extends string ? true : false;
IsString<string>;  // true
IsString<number>;  // false

// Extract types from arrays
type ArrayElement<T> = T extends (infer E)[] ? E : never;
type Num = ArrayElement<number[]>; // number

// Non-nullable
type NonNullable<T> = T extends null | undefined ? never : T;

TypeScript with React

// Component Props
interface ButtonProps {
  children: React.ReactNode;
  onClick?: () => void;
  variant?: "primary" | "danger";
  disabled?: boolean;
  className?: string;
}

const Button: React.FC<ButtonProps> = ({ children, onClick, variant = "primary", ...rest }) => (
  <button className={`btn btn-${variant}`} onClick={onClick} {...rest}>
    {children}
  </button>
);

// useState with type
const [user, setUser] = useState<User | null>(null);
const [items, setItems] = useState<string[]>([]);

// useRef with type
const inputRef = useRef<HTMLInputElement>(null);

// Event types
const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => { ... };
const handleSubmit = (e: React.FormEvent<HTMLFormElement>) => { ... };
const handleClick = (e: React.MouseEvent<HTMLButtonElement>) => { ... };
