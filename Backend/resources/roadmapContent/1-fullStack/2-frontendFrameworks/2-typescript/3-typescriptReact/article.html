TypeScript with React

TypeScript dramatically improves React development by catching prop mismatches, invalid event handlers, and incorrect hook usage at compile time rather than at runtime.

Typing Component Props

interface ButtonProps {
  variant?: 'primary' | 'secondary' | 'danger';
  size?: 'sm' | 'md' | 'lg';
  disabled?: boolean;
  onClick?: (event: React.MouseEvent<HTMLButtonElement>) => void;
  children: React.ReactNode;
  className?: string;
}

function Button({ variant = 'primary', size = 'md', children, onClick, disabled }: ButtonProps) {
  return (
    <button
      className={`btn btn--${variant} btn--${size}`}
      onClick={onClick}
      disabled={disabled}
    >
      {children}
    </button>
  );
}

// TypeScript catches mistakes at compile time:
<Button variant="invalid" /> // ❌ Error: not assignable to 'primary' | 'secondary' | 'danger'
<Button></Button>           // ❌ Error: 'children' is required

Typing useState and useReducer

// Explicit type when TS can't infer
const [user, setUser] = useState<User | null>(null);

// useReducer with discriminated union actions
type Action =
  | { type: 'INCREMENT' }
  | { type: 'DECREMENT' }
  | { type: 'RESET'; payload: number };

function reducer(state: number, action: Action): number {
  switch (action.type) {
    case 'INCREMENT': return state + 1;
    case 'DECREMENT': return state - 1;
    case 'RESET':     return action.payload;  // TS knows payload exists here
  }
}

Typing Event Handlers

// Input change
const handleChange = (event: React.ChangeEvent<HTMLInputElement>) => {
  setValue(event.target.value);
};

// Form submit
const handleSubmit = (event: React.FormEvent<HTMLFormElement>) => {
  event.preventDefault();
  // handle form data
};

// Button click
const handleClick = (event: React.MouseEvent<HTMLButtonElement>) => {
  // handle click
};

// Select change
const handleSelect = (event: React.ChangeEvent<HTMLSelectElement>) => {
  setOption(event.target.value);
};

Typing useRef

// DOM element ref
const inputRef = useRef<HTMLInputElement>(null);

// Safe access with optional chaining
const focusInput = () => inputRef.current?.focus();

// Mutable ref (not a DOM element)
const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

Typing Context

interface AuthContextType {
  user: User | null;
  login: (credentials: LoginCredentials) => Promise<void>;
  logout: () => void;
  isLoading: boolean;
}

// Non-null assertion because we always use this within the Provider
const AuthContext = createContext<AuthContextType>({} as AuthContextType);

// Typed consumer hook with guard
function useAuth(): AuthContextType {
  const context = useContext(AuthContext);
  if (!context) throw new Error('useAuth must be used within AuthProvider');
  return context;
}

Generic Components

// A reusable list component that works with any data type
interface ListProps<T> {
  items: T[];
  renderItem: (item: T, index: number) => React.ReactNode;
  keyExtractor: (item: T) => string | number;
  emptyMessage?: string;
}

function List<T>({ items, renderItem, keyExtractor, emptyMessage = 'No items' }: ListProps<T>) {
  if (!items.length) return <p>{emptyMessage}</p>;
  return (
    <ul>
      {items.map((item, index) => (
        <li key={keyExtractor(item)}>{renderItem(item, index)}</li>
      ))}
    </ul>
  );
}

// Usage
<List
  items={users}
  keyExtractor={u => u.id}
  renderItem={user => <UserCard user={user} />}
/>

Typing Custom Hooks

interface FetchState<T> {
  data: T | null;
  loading: boolean;
  error: string | null;
}

function useFetch<T>(url: string): FetchState<T> {
  const [state, setState] = useState<FetchState<T>>({
    data: null,
    loading: true,
    error: null
  });
  // ...effect
  return state;
}

// Usage — TS infers the type from the generic
const { data, loading } = useFetch<Product[]>('/api/products');

Best Practices

  • Use interface for component props and object shapes; type for unions and aliases.
  • Use discriminated unions for useReducer actions — TypeScript narrows the type in each case.
  • Enable "strict": true in tsconfig for maximum safety.
  • Avoid any — use unknown and type guard functions instead.
  • Make custom hooks generic when they work with multiple data types.
