Rate Limiting, Caching & API Security Hardening

Every public API faces abuse: scrapers, brute-force login attacks, DoS attempts, and resource-hungry clients making thousands of requests per second. Hardening your Node.js/Express API with rate limiting, intelligent caching, and a full security middleware stack is not optional in production.

Rate Limiting with express-rate-limit

import rateLimit from 'express-rate-limit';
import RedisStore from 'rate-limit-redis';
import { redisClient } from './redis';

// General API rate limit: 100 requests per 15 minutes per IP
export const apiLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,  // 15 minutes
  max: 100,
  standardHeaders: true,      // Return rate limit info in Retry-After headers
  legacyHeaders: false,
  // Redis store: shares rate limit state across multiple server instances
  store: new RedisStore({
    sendCommand: (...args) => redisClient.sendCommand(args),
  }),
  handler: (req, res) => {
    res.status(429).json({
      error: 'Too many requests',
      retryAfter: Math.ceil(req.rateLimit.resetTime / 1000),
    });
  },
});

// Stricter limit for auth endpoints (prevent brute force)
export const authLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 5,     // Only 5 login attempts per 15 minutes
  skipSuccessfulRequests: true,  // Don't count successful logins
  message: { error: 'Too many login attempts. Try again in 15 minutes.' },
});

// Apply in your app
app.use('/api/', apiLimiter);
app.use('/api/auth/login', authLimiter);
app.use('/api/auth/register', authLimiter);

Response Caching with Redis

import { redisClient } from './redis';

// Cache middleware factory
function cacheMiddleware(ttlSeconds: number) {
  return async (req, res, next) => {
    // Only cache GET requests
    if (req.method !== 'GET') return next();

    const cacheKey = `cache:${req.originalUrl}`;

    try {
      const cached = await redisClient.get(cacheKey);
      if (cached) {
        res.setHeader('X-Cache', 'HIT');
        return res.json(JSON.parse(cached));
      }
    } catch (err) {
      // Cache miss or Redis error — proceed to handler (don't break the request)
      console.error('Cache read error:', err);
    }

    // Intercept res.json to save response to cache
    const originalJson = res.json.bind(res);
    res.json = (data) => {
      redisClient.setEx(cacheKey, ttlSeconds, JSON.stringify(data))
        .catch(err => console.error('Cache write error:', err));
      res.setHeader('X-Cache', 'MISS');
      return originalJson(data);
    };

    next();
  };
}

// Cache products for 5 minutes, user-specific data for 30 seconds
app.get('/api/products', cacheMiddleware(300), getProducts);
app.get('/api/products/:id', cacheMiddleware(300), getProduct);

// Cache invalidation: when a product is updated, clear its cache
app.put('/api/products/:id', async (req, res) => {
  await updateProduct(req.params.id, req.body);
  await redisClient.del(`cache:/api/products/${req.params.id}`);
  await redisClient.del('cache:/api/products');  // Invalidate list too
  res.json({ success: true });
});

Helmet.js Security Headers

import helmet from 'helmet';

app.use(helmet({
  contentSecurityPolicy: {
    directives: {
      defaultSrc: ["'self'"],
      scriptSrc: ["'self'", "https://cdn.example.com"],
      styleSrc: ["'self'", "'unsafe-inline'"],
      imgSrc: ["'self'", "data:", "https:"],
      connectSrc: ["'self'", "https://api.example.com"],
    },
  },
  hsts: {
    maxAge: 31536000,          // 1 year
    includeSubDomains: true,
    preload: true,
  },
  frameguard: { action: 'deny' },          // Prevents clickjacking
  noSniff: true,                           // Prevent MIME type sniffing
  referrerPolicy: { policy: 'strict-origin-when-cross-origin' },
}));

// Request size limits (prevent large payload DoS)
app.use(express.json({ limit: '10kb' }));       // JSON body max 10KB
app.use(express.urlencoded({ limit: '10kb', extended: true }));

// CORS hardening
import cors from 'cors';
const allowedOrigins = process.env.ALLOWED_ORIGINS?.split(',') ?? [];

app.use(cors({
  origin: (origin, callback) => {
    if (!origin || allowedOrigins.includes(origin)) {
      callback(null, true);
    } else {
      callback(new Error(`Origin ${origin} not allowed by CORS`));
    }
  },
  credentials: true,
  methods: ['GET', 'POST', 'PUT', 'DELETE', 'PATCH'],
  allowedHeaders: ['Content-Type', 'Authorization'],
}));

SQL Injection & XSS Prevention

import DOMPurify from 'isomorphic-dompurify';
import { z } from 'zod';

// Always use parameterized queries (Prisma, pg with $1/$2 placeholders)
// NEVER: `SELECT * FROM users WHERE email = '${email}'` ← SQL Injection!
// ALWAYS: prisma.user.findUnique({ where: { email } }) ← safe

// Sanitize user-generated HTML content before storing/rendering
function sanitizeHtml(dirty: string): string {
  return DOMPurify.sanitize(dirty, {
    ALLOWED_TAGS: ['p', 'b', 'i', 'em', 'strong', 'a', 'ul', 'ol', 'li'],
    ALLOWED_ATTR: ['href', 'target'],
  });
}

// Validate + sanitize request body with Zod
const CreatePostSchema = z.object({
  title: z.string().min(1).max(200).trim(),
  content: z.string().min(1).max(10000).transform(sanitizeHtml),
  tags: z.array(z.string().max(50)).max(10),
});

app.post('/api/posts', async (req, res) => {
  const result = CreatePostSchema.safeParse(req.body);
  if (!result.success) {
    return res.status(400).json({ errors: result.error.format() });
  }
  // result.data is now validated and sanitized
  const post = await prisma.post.create({ data: result.data });
  res.status(201).json(post);
});
