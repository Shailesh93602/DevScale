Unit Testing with Jest

Unit tests verify that individual functions or components work correctly in isolation. They form the first layer of the testing pyramid and give you the fastest feedback — a well-tested codebase allows you to refactor with confidence, catch bugs before production, and treat tests as living documentation of expected behavior.

Jest Setup for a TypeScript Node.js Project

The AAA Pattern

Every unit test should be structured in three phases: Arrange (set up the test data and dependencies), Act (call the function under test), and Assert (verify the expected outcome):

// src/utils/pricing.ts
export function calculateDiscount(price: number, discountPct: number): number {
  if (price < 0) throw new Error('Price cannot be negative');
  if (discountPct < 0 || discountPct > 100) throw new Error('Invalid discount percentage');
  return Math.round(price * (1 - discountPct / 100) * 100) / 100;
}

// src/utils/pricing.test.ts
import { calculateDiscount } from './pricing';

describe('calculateDiscount', () => {
  it('applies a 20% discount to a price', () => {
    // Arrange
    const price = 100;
    const discountPct = 20;

    // Act
    const result = calculateDiscount(price, discountPct);

    // Assert
    expect(result).toBe(80);
  });

  it('rounds to 2 decimal places', () => {
    expect(calculateDiscount(9.99, 10)).toBe(8.99);
  });

  it('throws when price is negative', () => {
    expect(() => calculateDiscount(-10, 20)).toThrow('Price cannot be negative');
  });

  it('throws when discount exceeds 100%', () => {
    expect(() => calculateDiscount(100, 150)).toThrow('Invalid discount percentage');
  });

  it('returns full price for 0% discount', () => {
    expect(calculateDiscount(50, 0)).toBe(50);
  });
});

Mocking Dependencies

Unit tests test one unit in isolation. When your function depends on external services (database, API, email), you mock them to avoid real network calls and control the response:

// src/services/userService.ts
import { UserRepository } from '../repositories/userRepository';
import { EmailService } from '../services/emailService';

export class UserService {
  constructor(
    private userRepo: UserRepository,
    private emailSvc: EmailService
  ) {}

  async createUser(email: string, name: string) {
    const existing = await this.userRepo.findByEmail(email);
    if (existing) throw new Error('Email already registered');

    const user = await this.userRepo.create({ email, name });
    await this.emailSvc.sendWelcome(email, name);
    return user;
  }
}

// src/services/userService.test.ts
import { UserService } from './userService';
import { UserRepository } from '../repositories/userRepository';
import { EmailService } from '../services/emailService';

// Auto-mock the entire module
jest.mock('../repositories/userRepository');
jest.mock('../services/emailService');

describe('UserService.createUser', () => {
  let svc: UserService;
  let mockRepo: jest.Mocked;
  let mockEmail: jest.Mocked;

  beforeEach(() => {
    mockRepo = new UserRepository() as jest.Mocked;
    mockEmail = new EmailService() as jest.Mocked;
    svc = new UserService(mockRepo, mockEmail);
  });

  it('creates a user and sends a welcome email', async () => {
    // Arrange
    mockRepo.findByEmail.mockResolvedValue(null);  // No existing user
    mockRepo.create.mockResolvedValue({ id: '1', email: 'a@test.com', name: 'Alice' });
    mockEmail.sendWelcome.mockResolvedValue(undefined);

    // Act
    const user = await svc.createUser('a@test.com', 'Alice');

    // Assert
    expect(user.id).toBe('1');
    expect(mockRepo.create).toHaveBeenCalledWith({ email: 'a@test.com', name: 'Alice' });
    expect(mockEmail.sendWelcome).toHaveBeenCalledWith('a@test.com', 'Alice');
    expect(mockEmail.sendWelcome).toHaveBeenCalledTimes(1);
  });

  it('throws if email is already registered', async () => {
    mockRepo.findByEmail.mockResolvedValue({ id: '2', email: 'a@test.com', name: 'Bob' });

    await expect(svc.createUser('a@test.com', 'Alice'))
      .rejects.toThrow('Email already registered');

    // Verify no DB write or email was attempted
    expect(mockRepo.create).not.toHaveBeenCalled();
    expect(mockEmail.sendWelcome).not.toHaveBeenCalled();
  });
});

Testing React Components with React Testing Library

npm install --save-dev @testing-library/react @testing-library/user-event @testing-library/jest-dom
// components/LoginForm.test.tsx
import { render, screen } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import { LoginForm } from './LoginForm';

describe('LoginForm', () => {
  it('shows validation error when email is empty and form is submitted', async () => {
    const user = userEvent.setup();
    const mockSubmit = jest.fn();
    render(<LoginForm onSubmit={mockSubmit} />);

    // Act — click submit without filling in email
    await user.click(screen.getByRole('button', { name: /sign in/i }));

    // Assert — validation message appears
    expect(screen.getByText(/email is required/i)).toBeInTheDocument();
    expect(mockSubmit).not.toHaveBeenCalled();
  });

  it('calls onSubmit with credentials when form is valid', async () => {
    const user = userEvent.setup();
    const mockSubmit = jest.fn();
    render(<LoginForm onSubmit={mockSubmit} />);

    await user.type(screen.getByLabelText(/email/i), 'user@example.com');
    await user.type(screen.getByLabelText(/password/i), 'secret123');
    await user.click(screen.getByRole('button', { name: /sign in/i }));

    expect(mockSubmit).toHaveBeenCalledWith({ email: 'user@example.com', password: 'secret123' });
  });
});

Test Organization Best Practices

  • File location: Co-locate tests with the file they test (utils.ts → utils.test.ts in the same directory), not in a separate __tests__ folder. This makes it obvious when a file has no tests.
  • One assertion per test: When a test fails, a precise single assertion tells you exactly what broke. Multiple assertions mean one failing assertion silently skips the rest.
  • Test behavior, not implementation: Test that sendWelcomeEmail was called — not how the internal mailer object was constructed. Tests that depend on implementation details break when you refactor.
  • No magic numbers: Use named constants (const ADMIN_ROLE = 'admin') in both source and tests so the intent is clear when reading the test.
