Text Classification & Sentiment Analysis

Text classification assigns predefined categories to text — sentiment labels to reviews, intents to chatbot messages, topics to news articles. It's one of the most widely deployed NLP tasks in production. Modern approaches range from simple TF-IDF + logistic regression (fast, interpretable) to fine-tuned BERT models (state-of-the-art accuracy).

Baseline: TF-IDF + Logistic Regression

from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.linear_model import LogisticRegression
from sklearn.pipeline import Pipeline
from sklearn.model_selection import cross_val_score
import numpy as np

# TF-IDF: Term Frequency-Inverse Document Frequency
# Represents each document as a sparse vector of word importance scores
# Common words (the, is, a) get low weight (high DF); rare discriminative words get high weight

texts = ["The product quality is excellent!", "Terrible experience, never buying again"]
labels = [1, 0]  # 1=positive, 0=negative

# Build a production-quality baseline pipeline
baseline_pipeline = Pipeline([
    ('tfidf', TfidfVectorizer(
        max_features=50000,      # Vocabulary size cap
        ngram_range=(1, 2),      # Include bigrams (e.g., "not good" vs "good")
        min_df=3,                # Ignore terms appearing in < 3 documents
        max_df=0.95,             # Ignore terms in > 95% of documents (stop words effect)
        sublinear_tf=True,       # Log-scale TF: tf = 1 + log(tf) — reduces skew from common words
        analyzer='word',
        strip_accents='unicode'
    )),
    ('classifier', LogisticRegression(C=5.0, class_weight='balanced', max_iter=1000))
])

# Often achieves 85-90% accuracy on binary sentiment with minimal compute
cv_scores = cross_val_score(baseline_pipeline, texts, labels, cv=5, scoring='f1')
print(f"Baseline F1: {cv_scores.mean():.3f} ± {cv_scores.std():.3f}")
