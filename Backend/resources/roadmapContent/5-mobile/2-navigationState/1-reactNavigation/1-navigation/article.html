React Navigation: Screens, Stacks, Tabs & Drawers

Mobile apps don't have URLs — they have screens that animate in and out of view. React Navigation is the industry-standard navigation library for React Native, providing a fully customizable navigation architecture that maps closely to what iOS and Android users expect natively.

The NavigationContainer

Every React Navigation tree must be wrapped in a NavigationContainer at the root of your app. It manages the navigation state for the entire app:

// App.tsx
import { NavigationContainer } from '@react-navigation/native';
import { RootNavigator } from './navigation/RootNavigator';

export default function App() {
  return (
    <NavigationContainer>
      <RootNavigator />
    </NavigationContainer>
  );
}

Stack Navigation

The Stack Navigator mimics a physical stack of cards. Navigating to a new screen pushes a card on top; going back pops it off. This is the core pattern for hierarchical flows (list → detail → settings):

import { createNativeStackNavigator } from '@react-navigation/native-stack';
import HomeScreen from '../screens/HomeScreen';
import DetailsScreen from '../screens/DetailsScreen';

// Type-safe route params (TypeScript)
type RootStackParamList = {
  Home: undefined;
  Details: { productId: string; productName: string };
};

const Stack = createNativeStackNavigator<RootStackParamList>();

export function RootNavigator() {
  return (
    <Stack.Navigator
      initialRouteName="Home"
      screenOptions={{
        headerStyle: { backgroundColor: '#6366f1' },
        headerTintColor: '#fff',
        headerTitleStyle: { fontWeight: 'bold' },
        animation: 'slide_from_right', // iOS natural feel
      }}
    >
      <Stack.Screen name="Home" component={HomeScreen} options={{ title: 'Product List' }} />
      <Stack.Screen
        name="Details"
        component={DetailsScreen}
        options={({ route }) => ({ title: route.params.productName })} // Dynamic title
      />
    </Stack.Navigator>
  );
}
// HomeScreen.tsx — navigating with typed params
import { NativeStackScreenProps } from '@react-navigation/native-stack';
import { RootStackParamList } from '../navigation/RootNavigator';

type Props = NativeStackScreenProps<RootStackParamList, 'Home'>;

export default function HomeScreen({ navigation }: Props) {
  return (
    <Button
      title="See Details"
      onPress={() => navigation.navigate('Details', { productId: '42', productName: 'Hiking Boots' })}
    />
  );
}

// DetailsScreen.tsx — reading params
type Props = NativeStackScreenProps<RootStackParamList, 'Details'>;
export default function DetailsScreen({ route, navigation }: Props) {
  const { productId, productName } = route.params;
  return (
    <View>
      <Text>{productName} (ID: {productId})</Text>
      <Button title="Go Back" onPress={() => navigation.goBack()} />
    </View>
  );
}

Bottom Tab Navigation

Bottom Tabs are the primary navigation pattern of modern mobile apps (Instagram, Twitter). Each tab maintains its own independent navigation stack:

import { createBottomTabNavigator } from '@react-navigation/bottom-tabs';
import Ionicons from '@expo/vector-icons/Ionicons';

const Tab = createBottomTabNavigator();

export function TabNavigator() {
  return (
    <Tab.Navigator
      screenOptions={({ route }) => ({
        tabBarIcon: ({ focused, color, size }) => {
          // Return a vector icon based on route name and focus state
          const iconMap = {
            Home: focused ? 'home' : 'home-outline',
            Search: focused ? 'search' : 'search-outline',
            Profile: focused ? 'person' : 'person-outline',
          };
          return <Ionicons name={iconMap[route.name]} size={size} color={color} />;
        },
        tabBarActiveTintColor: '#6366f1',
        tabBarInactiveTintColor: 'gray',
      })}
    >
      <Tab.Screen name="Home" component={HomeStack} />
      <Tab.Screen name="Search" component={SearchScreen} />
      <Tab.Screen name="Profile" component={ProfileScreen} />
    </Tab.Navigator>
  );
}

Drawer Navigation

import { createDrawerNavigator } from '@react-navigation/drawer';

const Drawer = createDrawerNavigator();

export function DrawerNavigator() {
  return (
    <Drawer.Navigator
      screenOptions={{
        drawerStyle: { backgroundColor: '#1e1b4b', width: 240 },
        drawerLabelStyle: { color: '#fff' },
        drawerActiveTintColor: '#818cf8',
      }}
    >
      <Drawer.Screen name="Home" component={HomeScreen} />
      <Drawer.Screen name="Settings" component={SettingsScreen} />
      <Drawer.Screen name="About" component={AboutScreen} />
    </Drawer.Navigator>
  );
}

Deep Linking

Deep links let users open a specific screen from a URL (e.g., from a push notification or web link):

// In NavigationContainer, configure link mapping
const linking = {
  prefixes: ['myapp://', 'https://myapp.com'],
  config: {
    screens: {
      Home: '',
      Details: 'product/:productId',
      Profile: 'user/:userId',
    },
  },
};

// <NavigationContainer linking={linking}>

useNavigation & useRoute Hooks

When passing navigation prop everywhere becomes tedious, use hooks inside deeply nested child components:

import { useNavigation, useRoute } from '@react-navigation/native';

function NestedButton() {
  const navigation = useNavigation();
  return <Button title="Go Home" onPress={() => navigation.navigate('Home')} />;
}

function NestedComponent() {
  const route = useRoute();
  return <Text>Current screen: {route.name}</Text>;
}

Best Practices

  • Always define a typed ParamList in TypeScript — it gives you autocomplete for params and catches route name typos at compile time.
  • Nest navigators deliberately: Tab inside Stack for post-auth flows, Stack inside each Tab for drill-down navigation.
  • Use navigation.replace('Home') instead of navigate when you don't want the user to be able to press Back (e.g., after successful login).
  • Don't put business logic in _layout.tsx — keep navigators as pure routing configuration, not data fetching containers.
