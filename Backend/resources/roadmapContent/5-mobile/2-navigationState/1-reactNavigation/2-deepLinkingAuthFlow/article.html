Deep Linking & Authentication Flow

Deep links route external URLs directly into specific screens of your app — enabling marketing campaigns, email magic links, and shared content URLs. A robust authentication flow correctly handles unauthenticated users, loading states, and persistent sessions without exposing protected screens.

Deep Link Setup

// app.json — configure URL scheme and universal links
{
  "expo": {
    "scheme": "myapp",           // myapp://  — custom URL scheme
    "ios": {
      "associatedDomains": ["applinks:myapp.com"]  // Universal Links (https://)
    },
    "android": {
      "intentFilters": [{
        "action": "VIEW",
        "autoVerify": true,
        "data": [{ "scheme": "https", "host": "myapp.com" }],
        "category": ["BROWSABLE", "DEFAULT"]
      }]
    }
  }
}

// React Navigation deep link configuration
import { NavigationContainer } from '@react-navigation/native';

const linking = {
  prefixes: ['myapp://', 'https://myapp.com'],
  config: {
    screens: {
      // Stack navigator
      Home: 'home',
      // myapp://product/123 → ProductDetail screen with id=123
      ProductDetail: 'product/:id',
      // myapp://profile?tab=orders → Profile with tab param
      Profile: {
        path: 'profile',
        parse: { tab: (tab) => tab },
      },
      // Nested navigators
      Tabs: {
        screens: {
          Orders: 'orders',
          Settings: 'settings',
        },
      },
    },
  },
};

export default function App() {
  return (
    <NavigationContainer linking={linking}>
      <RootNavigator />
    </NavigationContainer>
  );
}

The Authentication Navigation Pattern

import { createNativeStackNavigator } from '@react-navigation/native-stack';
import { useAuthStore } from './store/authStore';

const Stack = createNativeStackNavigator();

function RootNavigator() {
  const { user, isLoading } = useAuthStore();

  // Show splash/loading screen while checking stored session
  if (isLoading) return <SplashScreen />;

  return (
    <Stack.Navigator screenOptions={{ headerShown: false }}>
      {user ? (
        // User is signed in — show app screens
        // React Navigation automatically transitions from Auth → App group
        <>
          <Stack.Screen name="Home" component={HomeScreen} />
          <Stack.Screen name="Profile" component={ProfileScreen} />
          <Stack.Screen name="Settings" component={SettingsScreen} />
        </>
      ) : (
        // No user — show auth screens
        // Protected screens are COMPLETELY absent from navigator tree
        // Deep links to protected screens are queued and resolved after login
        <>
          <Stack.Screen name="Welcome" component={WelcomeScreen} />
          <Stack.Screen name="Login" component={LoginScreen} />
          <Stack.Screen name="Signup" component={SignupScreen} />
        </>
      )}
    </Stack.Navigator>
  );
}

Session Persistence with SecureStore

import * as SecureStore from 'expo-secure-store';

// On login: save session token securely (iOS Keychain / Android Keystore)
await SecureStore.setItemAsync('session_token', token);

// On app launch: restore session  
const token = await SecureStore.getItemAsync('session_token');
if (token) {
  // Validate token with server, then set user in state
  const user = await validateToken(token);
  authStore.setUser(user);
}

// On logout: clear all secure storage
await SecureStore.deleteItemAsync('session_token');
